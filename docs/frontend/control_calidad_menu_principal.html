 54    def setImage(self, image_path):
 55        image = QImage(image_path)
 56        if image.isNull():
 57            print(f"\u274c Error: no se pudo cargar la imagen desde {image_path}")
 58            return False
 59        self.pixmap = QPixmap.fromImage(image)
 60        self.is_showing_message = False
 61        self.updateImage()
 62        return True
 63
 64    def updateImage(self):
 65        if not self.pixmap or self.pixmap.isNull():
 66            return
 67        self.scene.clear()
 68        self.image_item = self.scene.addPixmap(self.pixmap)
 69        # Usamos el tamaño real de la imagen como rectángulo de escena
 70        self.scene.setSceneRect(self.image_item.boundingRect())
 71        # Ajustamos la vista respetando proporción
 72        self.fitInView(self.image_item, Qt.KeepAspectRatio)
 73        self.text_item = None
 74
 75    def showMessage(self, message, bgColor="#2C7873", textColor="#FFFFFF"):
 76        self.scene.clear()
 77        self.image_item = None
 78        self.is_showing_message = True
 79        self.current_message = message
 80        self.current_bg_color = bgColor
 81        self.current_text_color = textColor
 82        
 83        self._updateMessageDisplay()
 84
 85    def _updateMessageDisplay(self):
 86        if not self.is_showing_message:
 87            return
 88            
 89        # Obtener el tamaño actual de la vista
 90        view_width = max(1, self.viewport().width())
 91        view_height = max(1, self.viewport().height())
 92        
 93        # Actualizar el rectángulo de la escena
 94        self.scene.clear()
 95        self.scene.setSceneRect(0, 0, view_width, view_height)
 96        
 97        # Fondo del visor completo
 98        rect = QRectF(0, 0, view_width, view_height)
 99        self.scene.addRect(rect, brush=QBrush(QColor(self.current_bg_color)))
100        
101        # Texto proporcional - ajustamos el tamaño de la fuente según las dimensiones actuales
102        font_size = max(10, int(min(view_width, view_height) * 0.08))
103        font = QFont("Arial", font_size, QFont.Bold)
104        
105        self.text_item = QGraphicsTextItem(self.current_message)
106        self.text_item.setFont(font)
107        self.text_item.setDefaultTextColor(QColor(self.current_text_color))
108        self.scene.addItem(self.text_item)
109        
110        # Centrar texto
111        text_rect = self.text_item.boundingRect()
112        x = (view_width - text_rect.width()) / 2
113        y = (view_height - text_rect.height()) / 2
114        self.text_item.setPos(x, y)
115        
116        # Ajustar la vista al contenido
117        self.fitInView(self.scene.sceneRect(), Qt.IgnoreAspectRatio)
118
119    def resizeEvent(self, event):
120        super().resizeEvent(event)
121        
122        if self.is_showing_message:
123            # Si estamos mostrando un mensaje, actualizamos su display
124            self._updateMessageDisplay()
125        elif self.image_item:
126            # Si estamos mostrando una imagen, la ajustamos
127            self.fitInView(self.image_item, Qt.KeepAspectRatio)

363        """
364        Ejecuta el análisis de defectos sobre las imágenes del rollo seleccionado.
365
366        Procesa las imágenes mediante la función `analizar_rollo`,

369        if self.timer and self.timer.isActive():

371            return

374
375        if seleccion == "-- Seleccione un rollo --":
376            QMessageBox.warning(self, "Advertencia", "Por favor, seleccione una carpeta primero.")
377            return
378
379        self.folder = os.path.join(self.base_folder, seleccion)
380        umbral_usuario = float(self.ui.doubleSpinBox.value())
381        self.ui.label_contador.setText("📊 0 / {}".format(len(self.images)))
382
383        # Ejecutar el análisis de imágenes antes de cargarlas
384        try:

389            return
390
391        # Cargar imágenes desde 'originales' y 'procesado'
392        carpeta_originales = os.path.join(self.folder, "originales")

395        self.imagenes_procesadas = self.cargar_imagenes(carpeta_procesadas)
396
397        # Validar que ambos conjuntos estén sincronizados
398        if len(self.imagenes_originales) != len(self.imagenes_procesadas):
399            QMessageBox.warning(self, "Advertencia", "El número de imágenes originales y procesadas no coincide.")
400            return
401
402        self.images = list(zip(self.imagenes_originales, self.imagenes_procesadas))
403        self.index = 0
404        self.imagenes_procesadas = []  # Limpiar acumulación previa
405        self.analisis_completado = False
406
407        self.ui.tableWidget.setRowCount(0)
408        self.ui.progressBar.setMaximum(len(self.images) if self.images else 1)
409        self.ui.progressBar.setValue(0)
410
411        if not self.images:

414            self.ui.label_5.setText("Sin imágenes")

416            return
417
418        self.timer = QTimer()
419        self.timer.timeout.connect(self.mostrar_siguiente_imagen)

574        """
575        Lee el archivo .txt correspondiente a una imagen procesada.
576        Devuelve:
577        - área mínima
578        - área máxima
579        - lista completa de áreas

581        try:
582            txt_path = ruta_imagen_procesada + ".txt"
583            if not os.path.exists(txt_path):
584                return None, None, []
585
586            areas = []
587            with open(txt_path, "r", encoding="utf-8") as f:
588                for line in f:
589                    partes = line.strip().split()
590                    if len(partes) == 2 and partes[1].endswith("mm2"):
591                        valor = float(partes[1].replace("mm2", ""))
592                        areas.append(valor)
593
594            if not areas:
595                return None, None, []
596
597            return min(areas), max(areas), areas

574    def leer_defectos_txt(self, ruta_imagen_procesada):
575        """
576        Lee el archivo .txt correspondiente a una imagen procesada.
577        Devuelve:
578        - área mínima
579        - área máxima
580        - lista completa de áreas
581        """
582        try:
583            txt_path = ruta_imagen_procesada + ".txt"
584            if not os.path.exists(txt_path):
585                return None, None, []
586
587            areas = []
588            with open(txt_path, "r", encoding="utf-8") as f:
589                for line in f:
590                    partes = line.strip().split()
591                    if len(partes) == 2 and partes[1].endswith("mm2"):
592                        valor = float(partes[1].replace("mm2", ""))
593                        areas.append(valor)
594
595            if not areas:
596                return None, None, []
597
598            return min(areas), max(areas), areas
599        except Exception as e:
600            print(f"Error al leer defectos desde TXT: {e}")
601            return None, None, []
