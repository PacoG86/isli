116    def create_bimodal_histogram(image: np.ndarray[np.uint8], peak1: int = 20, peak2: int = 240, weight1: float = 0.99, weight2: float = 0.01) -> np.ndarray[np.uint8]:
117        """
118        Transforma una imagen en escala de grises para generar una distribución bimodal de intensidades.
119
120        Esta transformación sirve para aumentar el contraste de la imagen antes de aplicar técnicas
121        de segmentación. Utiliza dos curvas gaussianas con pesos diferentes para modificar la 
122        distribución de histograma.
123
124        Args:
125            image (np.ndarray): Imagen de entrada en escala de grises.
126            peak1 (int): Pico de intensidad principal de la primera gaussiana (más oscura). Por defecto 20.
127            peak2 (int): Pico de intensidad de la segunda gaussiana (más clara). Por defecto 240.
128            weight1 (float): Peso relativo de la primera gaussiana. Por defecto 0.99.
129            weight2 (float): Peso relativo de la segunda gaussiana. Por defecto 0.01.
130
131        Raises:
132            ValueError: Si la imagen no está en escala de grises.
133
134        Returns:
135            np.ndarray: Imagen resultante con distribución de intensidades bimodal.
136        """
137        if len(image.shape) > 2:
138            raise ValueError("Input image must be a grayscale image.")
139
140        normalized_image = image / 255.0
141
142        transformation = np.zeros(256, dtype=np.float32)
143        for i in range(256):
144            t1 = weight1 * np.exp(-((i - peak1) ** 2) / (2 * (30 ** 2))) 
145            t2 = weight2 * np.exp(-((i - peak2) ** 2) / (2 * (60 ** 2))) 
146            transformation[i] = t1 + t2
147
148        transformation /= np.sum(transformation)
149
150        cdf = np.cumsum(transformation)
151        cdf_normalized = cdf / cdf[-1]
152
153        output_image = np.interp(normalized_image.ravel(), np.linspace(0, 1, 256), cdf_normalized)
154        output_image = (output_image * 255).astype('uint8').reshape(image.shape)
155
156        return output_image

233        """
234        Aplica umbralización adaptativa para segmentar defectos oscuros (blackspots) en una imagen en escala de grises.
235
236        Esta técnica ajusta el umbral dinámicamente en función de las regiones locales de la imagen,
237        lo que permite una detección más precisa en condiciones de iluminación no uniforme.
238
239        Args:
240            image (np.ndarray[np.uint8]): Imagen de entrada en escala de grises.
241
242        Returns:
243            np.ndarray[np.bool_]: Máscara binaria donde los píxeles `True` indican regiones con posibles defectos.
244        
245        Notas:
246            - El tamaño del bloque para la umbralización se calcula como la mitad del menor lado de la imagen,
247            y se ajusta para que siempre sea impar.
248            - Se aplica un desenfoque Gaussiano previo para reducir ruido.
249        """
250
251        blur = cv2.GaussianBlur(image, (5,5), 0)
252        
253        block_size = min(image.shape[0], image.shape[1]) // 2
254        #block_size = min(image.shape[0], image.shape[1])
255        if block_size % 2 == 0:
256            block_size -= 1
257
258        blackspot_mask = cv2.adaptiveThreshold(blur,255,cv2.ADAPTIVE_THRESH_GAUSSIAN_C,\
259                cv2.THRESH_BINARY_INV,block_size,2)
260
261        return blackspot_mask.astype(np.bool)

265        """

287        return blackspot_mask.astype(np.bool)

116    @staticmethod
117    def create_bimodal_histogram(image: np.ndarray[np.uint8], peak1: int = 20, peak2: int = 240, weight1: float = 0.99, weight2: float = 0.01) -> np.ndarray[np.uint8]:
118        """
119        Transforma una imagen en escala de grises para generar una distribución bimodal de intensidades.
120
121        Esta transformación sirve para aumentar el contraste de la imagen antes de aplicar técnicas
122        de segmentación. Utiliza dos curvas gaussianas con pesos diferentes para modificar la 
123        distribución de histograma.
124
125        Args:
126            image (np.ndarray): Imagen de entrada en escala de grises.
127            peak1 (int): Pico de intensidad principal de la primera gaussiana (más oscura). Por defecto 20.
128            peak2 (int): Pico de intensidad de la segunda gaussiana (más clara). Por defecto 240.
129            weight1 (float): Peso relativo de la primera gaussiana. Por defecto 0.99.
130            weight2 (float): Peso relativo de la segunda gaussiana. Por defecto 0.01.
131
132        Raises:
133            ValueError: Si la imagen no está en escala de grises.
134
135        Returns:
136            np.ndarray: Imagen resultante con distribución de intensidades bimodal.
137        """
138        if len(image.shape) > 2:
139            raise ValueError("Input image must be a grayscale image.")
140
141        normalized_image = image / 255.0
142
143        transformation = np.zeros(256, dtype=np.float32)
144        for i in range(256):
145            t1 = weight1 * np.exp(-((i - peak1) ** 2) / (2 * (30 ** 2))) 
146            t2 = weight2 * np.exp(-((i - peak2) ** 2) / (2 * (60 ** 2))) 
147            transformation[i] = t1 + t2
148
149        transformation /= np.sum(transformation)
150
151        cdf = np.cumsum(transformation)
152        cdf_normalized = cdf / cdf[-1]
153
154        output_image = np.interp(normalized_image.ravel(), np.linspace(0, 1, 256), cdf_normalized)
155        output_image = (output_image * 255).astype('uint8').reshape(image.shape)
156
157        return output_image

123        distribución de histograma.

140

142
143        transformation = np.zeros(256, dtype=np.float32)
144        for i in range(256):
145            t1 = weight1 * np.exp(-((i - peak1) ** 2) / (2 * (30 ** 2))) 
146            t2 = weight2 * np.exp(-((i - peak2) ** 2) / (2 * (60 ** 2))) 
147            transformation[i] = t1 + t2

157        return output_image
